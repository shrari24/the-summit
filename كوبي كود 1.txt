            <h1 class="text-7xl font-black italic mb-2 text-white uppercase">القمة</h1>

                <input id="name-in" type="text" placeholder="اسمك..." class="bg-slate-900 border border-slate-700 p-4 rounded-xl text-center font-bold">
                <button onclick="start()" class="py-6 bg-blue-600 rounded-3xl font-black text-2xl shadow-xl">ابدأ الآن</button>

            <div class="flex justify-between items-start">

                <div class="glass px-4 py-2 font-black text-xl text-yellow-500"><span id="gold">0</span> 🪙</div>
            </div>
            <div id="jump-ui" class="absolute inset-0 flex items-center justify-center interactive hidden">
                <button onclick="jump()" class="px-16 py-6 bg-yellow-500 text-black font-black text-2xl rounded-full animate-bounce">اقفز! 🪂</button>
            </div>
            <div id="ctrl" class="hidden w-full h-full relative interactive">
                <div id="joystick-base"><div id="joystick-red"></div></div>
                <div class="absolute bottom-10 right-10 flex flex-col gap-4">
                    <button onmousedown="shooting=true" onmouseup="shooting=false" class="btn-act bg-red-600 border-none">🔥</button>
                    <button onclick="build()" class="btn-act bg-blue-600 border-none">🧱</button>
                </div>

    <script>
        let scene, camera, renderer, player, plane, clock;
        let gameState = 'MENU', gold = 0, enemies = [], bullets = [], joy = {x:0, y:0}, shooting = false;
        function init() {
            scene = new THREE.Scene(); scene.background = new THREE.Color(0x020617);
            camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            document.body.appendChild(renderer.domElement);
            scene.add(new THREE.HemisphereLight(0xffffff, 0x444444, 1.2));
            const sun = new THREE.DirectionalLight(0xffffff, 1); sun.position.set(50, 200, 50); scene.add(sun);
            const grid = new THREE.GridHelper(1000, 40, 0x1e293b, 0x0f172a); scene.add(grid);
            clock = new THREE.Clock(); animate();
        }
        function start() {
            gameState = 'PLANE';
            document.getElementById('menu').style.display = 'none';
            document.getElementById('hud').classList.remove('hidden');
            document.getElementById('jump-ui').classList.remove('hidden');
            plane = new THREE.Mesh(new THREE.BoxGeometry(10,5,25), new THREE.MeshPhongMaterial({color: 0x334155}));
            plane.position.set(0, 300, 500); scene.add(plane);
            player = new THREE.Mesh(new THREE.BoxGeometry(2,4,2), new THREE.MeshPhongMaterial({color: 0x3b82f6}));
            scene.add(player);
        }
        function jump() {
            gameState = 'FALLING'; document.getElementById('jump-ui').style.display = 'none';
            player.position.copy(plane.position); player.add(camera); camera.position.set(0,5,12);
        }
        function build() {
            const w = new THREE.Mesh(new THREE.BoxGeometry(8,6,1), new THREE.MeshPhongMaterial({color: 0x3b82f6, transparent: true, opacity: 0.6}));
            const d = new THREE.Vector3(0,0,-1).applyQuaternion(player.quaternion);
            w.position.copy(player.position).add(d.multiplyScalar(8)); w.position.y=3;
            w.quaternion.copy(player.quaternion); scene.add(w);
        }
        function animate() {
            requestAnimationFrame(animate); const dt = clock.getDelta();
            if(gameState === 'PLANE') {
                plane.position.z -= 1.5; camera.position.set(plane.position.x, plane.position.y+40, plane.position.z+80); camera.lookAt(plane.position);
            } else if(gameState === 'FALLING') {
                player.position.y -= 2; if(player.position.y <= 2) {
                    gameState = 'GROUND'; player.position.y = 2; document.getElementById('ctrl').classList.remove('hidden');
                    for(let i=0; i<50; i++) {
                        const en = new THREE.Mesh(new THREE.BoxGeometry(2,4,2), new THREE.MeshPhongMaterial({color: 0xef4444}));
                        en.position.set((Math.random()-0.5)*800, 2, (Math.random()-0.5)*800); scene.add(en); enemies.push(en);
                    }
                }
            } else if(gameState === 'GROUND') {
                player.rotation.y -= joy.x * 0.05; player.translateZ(-joy.y * 1);
                if(shooting) { /* إطلاق النار */ }
            }
            renderer.render(scene, camera);
        }
        const b = document.getElementById('joystick-base'), s = document.getElementById('joystick-red'); let act = false;
        b.addEventListener('touchstart', (e) => { act = true; e.preventDefault(); });
        window.addEventListener('touchmove', (e) => {
            if(!act) return; const r = b.getBoundingClientRect(), cx = r.left + r.width/2, cy = r.top + r.height/2;
            const tx = e.touches[0].clientX - cx, ty = e.touches[0].clientY - cy, d = Math.min(Math.sqrt(tx*tx+ty*ty), 35);
            const a = Math.atan2(ty, tx); s.style.transform = `translate(${Math.cos(a)*d}px, ${Math.sin(a)*d}px)`;
            joy = { x: (Math.cos(a)*d)/35, y: (Math.sin(a)*d)/35 };
        });
        window.addEventListener('touchend', () => { act = false; s.style.transform = ''; joy = {x:0, y:0}; });
        window.onload = init;
    </script>
